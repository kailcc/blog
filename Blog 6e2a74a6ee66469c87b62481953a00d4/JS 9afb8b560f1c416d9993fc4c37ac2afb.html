JS考题

下面会输出什么

var y = 1;

if (function f(){}) {
  y += typeof f;
}

console.log(y);
  • 解

    输出是"1undefined"， if条件语句使用eval进行计算，因此eval(function f(){})

    会返回f(){},当作表达式执行，所以typeof f === undefined

实现一个异步求和函数

提供一个异步add方法如下，需要实现一个await sum(...args)函数;

function asyncAdd(a, b, callback) {
  setTimeout(function () {
    callback(null, a + b);
  }, 1000);
}
  • 解
    async function sum(...args) {
	return args.reduce((promise, cur) => {
		return promise.then((acc) => new Promise((resolve) => {
			asyncAdd(acc, cur, (_, val) => {
				resolve(val)
			})
		}))
	}, Promise.resolve(0))
}

    用分治思想再优化

    
async function sum(...args) {
	if (args.length === 1) return args[0]

	const promises = []

	for (let i = 0; i < args.length; i += 2) {
		promises.push(new Promise((resolve) => {
			asyncAdd(args[i], args[i + 1], (_, val) => {
				resolve(val)
			})
		}))
	}

	return sum(...await Promise.all(promises))
}

    Promise.all会导致快的要等慢的，如果异步加法耗时不是固定的，怎么优化呢？

    async function sum(...args) {
	if (args.length === 1) return args[0]

	const promises = []

	for (let i = 0; i < args.length; i += 2) {
		const p1 = typeof args[i] === 'number' ? Promise.resolve(args[i]) : args[i]
		const p2 = typeof args[i + 1] === 'number' ? Promise.resolve(args[i + 1]) : args[i + 1]

		promises.push(
			Promise.all([p1, p2]).then(([a, b]) => {
				return new Promise((resolve) => {
					asyncAdd(a, b, (_, val) => {
						resolve(val)
					})
				})
			})
		)
	}

	return sum(...promises)
}
