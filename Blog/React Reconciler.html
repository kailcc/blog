React Reconciler

在React v16版本之前，React使用的是Stack Reconciler，顾名思义，在我们调用setState时，React会遍历应用的所有节点，更新State和Props，调用LifeCycle Function，Diff，计算更新，最后再更新DOM。整个过程一气呵成，不能被打断。导致大量同步任务阻塞了浏览器的UI渲染。默认情况下，JS执行，UI渲染都是互斥的运行在浏览器的主线程当中。倘若JS执行占用主线程超过16ms，就会出现掉帧的情况。

Fiber Reconciler

解决主线程长时间被占用这一问题的基本思路就是Time-slicing，时间分片是将大计算量的任务，分割成几个片段去完成。也就是说完成一个片段之后，将控制权交还给浏览器，让浏览器调度各线程工作。等浏览器完成工作之后，再继续完成下一个片段。

要实现暂停遍历并阻止堆栈增长，可以使用单链表树遍历算法。

Fiber 数据结构

  • stateNode — 引用实例
  • child — 引用第一个子节点
  • sibling — 引用第一个相邻节点
  • return — 引用父节点

用代码表示上图节点间的关系，比如a1下面有b1、b2、b3

var a1 = { name: 'a1', render: () => [b1, b2, b3] }
var b1 = { name: 'b1', render: () => [] }
var b2 = { name: 'b2', render: () => [c1] }
var b3 = { name: 'b3', render: () => [c2] }
var c1 = { name: 'c1', render: () => [d1, d2] }
var c2 = { name: 'c2', render: () => [] }
var d1 = { name: 'd1', render: () => [] }
var d2 = { name: 'd2', render: () => [] }

构建节点的数据结构, 如下所示

function FiberNode(instance) {
	this.stateNode = instance
	this.return = null
	this.sibling = null
  this.child = null
}

创建a1节点的Fiber结构

function connect(parent, childList) {
	parent.child = childList.reduceRigth((prev, current) => {
		const fiberNode = new FiberNode(current)
		fiberNode.sibling = prev
		fiberNode.return = parent

		return fiberNode
	}, null)

	return parent.child
}

var parent = new FiberNode(a1)
var firstChild = connect(parent, a1.render())

这样就完成了a1 → b1，b1→b2→b3，以及b1、b2、b3都→a1的链表结构

最后完成walk

function doWork(node) {
	console.log(node.stateNode.name)
	const childList = node.stateNode.render()

	if (childList.length > 0) {
		return connect(node, childList)
	}
}

function walk(root) {
	let currentNode = root
	
	while(true) {
		const child = doWork(currentNode)
		
		if (child) {
			currentNode = child
      continue
		}

		while(!currentNode.sibling) {
			currentNode = currentNode.return
      if (currentNode === root) {
        return
      }
		}

		currentNode = currentNode.sibling
	}
}

walk(new FiberNode(a1))
